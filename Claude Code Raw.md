‚îÇ ‚îÇ     def _extract_imports(self, file_path, content):                                                                                                                                                                                                                              ‚îÇ ‚îÇ
‚îÇ ‚îÇ         """Extract imports using multiple methods"""                                                                                                                                                                                                                             ‚îÇ ‚îÇ
‚îÇ ‚îÇ         imports = []                                                                                                                                                                                                                                                             ‚îÇ ‚îÇ
‚îÇ ‚îÇ                                                                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
‚îÇ ‚îÇ         # Method 1: Regex patterns                                                                                                                                                                                                                                               ‚îÇ ‚îÇ
‚îÇ ‚îÇ         import_patterns = [                                                                                                                                                                                                                                                      ‚îÇ ‚îÇ
‚îÇ ‚îÇ             r'^import\s+([^\s#]+)',                                                                                                                                                                                                                                              ‚îÇ ‚îÇ
‚îÇ ‚îÇ             r'^from\s+([^\s#]+)\s+import\s+([^#]+)',                                                                                                                                                                                                                             ‚îÇ ‚îÇ
‚îÇ ‚îÇ         ]                                                                                                                                                                                                                                                                        ‚îÇ ‚îÇ
‚îÇ ‚îÇ                                                                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
‚îÇ ‚îÇ         for line_num, line in enumerate(content.splitlines(), 1):                                                                                                                                                                                                                ‚îÇ ‚îÇ
‚îÇ ‚îÇ             line = line.strip()                                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
‚îÇ ‚îÇ                                                                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
‚îÇ ‚îÇ             for pattern in import_patterns:                                                                                                                                                                                                                                      ‚îÇ ‚îÇ
‚îÇ ‚îÇ                 match = re.match(pattern, line)                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
‚îÇ ‚îÇ                 if match:                                                                                                                                                                                                                                                        ‚îÇ ‚îÇ
‚îÇ ‚îÇ                     if 'from' in pattern:                                                                                                                                                                                                                                        ‚îÇ ‚îÇ
‚îÇ ‚îÇ                         module = match.group(1)                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
‚îÇ ‚îÇ                         items = match.group(2)                                                                                                                                                                                                                                   ‚îÇ ‚îÇ
‚îÇ ‚îÇ                         imports.append({                                                                                                                                                                                                                                         ‚îÇ ‚îÇ
‚îÇ ‚îÇ                             'type': 'from_import',                                                                                                                                                                                                                               ‚îÇ ‚îÇ
‚îÇ ‚îÇ                             'module': module,                                                                                                                                                                                                                                    ‚îÇ ‚îÇ
‚îÇ ‚îÇ                             'items': [item.strip() for item in items.split(',')],                                                                                                                                                                                                ‚îÇ ‚îÇ
‚îÇ ‚îÇ                             'line': line_num,                                                                                                                                                                                                                                    ‚îÇ ‚îÇ
‚îÇ ‚îÇ                             'raw': line                                                                                                                                                                                                                                          ‚îÇ ‚îÇ
‚îÇ ‚îÇ                         })                                                                                                                                                                                                                                                       ‚îÇ ‚îÇ
‚îÇ ‚îÇ                     else:                                                                                                                                                                                                                                                        ‚îÇ ‚îÇ
‚îÇ ‚îÇ                         module = match.group(1)                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
‚îÇ ‚îÇ                         imports.append({                                                                                                                                                                                                                                         ‚îÇ ‚îÇ
‚îÇ ‚îÇ                             'type': 'import',                                                                                                                                                                                                                                    ‚îÇ ‚îÇ
‚îÇ ‚îÇ                             'module': module,                                                                                                                                                                                                                                    ‚îÇ ‚îÇ
‚îÇ ‚îÇ                             'line': line_num,                                                                                                                                                                                                                                    ‚îÇ ‚îÇ
‚îÇ ‚îÇ                             'raw': line                                                                                                                                                                                                                                          ‚îÇ ‚îÇ
‚îÇ ‚îÇ                         })                                                                                                                                                                                                                                                       ‚îÇ ‚îÇ
‚îÇ ‚îÇ                     break                                                                                                                                                                                                                                                        ‚îÇ ‚îÇ
‚îÇ ‚îÇ                                                                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
‚îÇ ‚îÇ         # Method 2: AST parsing (more reliable)                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
‚îÇ ‚îÇ         try:                                                                                                                                                                                                                                                                     ‚îÇ ‚îÇ
‚îÇ ‚îÇ             tree = ast.parse(content)                                                                                                                                                                                                                                            ‚îÇ ‚îÇ
‚îÇ ‚îÇ             for node in ast.walk(tree):                                                                                                                                                                                                                                          ‚îÇ ‚îÇ
‚îÇ ‚îÇ                 if isinstance(node, ast.Import):                                                                                                                                                                                                                                 ‚îÇ ‚îÇ
‚îÇ ‚îÇ                     for alias in node.names:                                                                                                                                                                                                                                     ‚îÇ ‚îÇ
‚îÇ ‚îÇ                         imports.append({                                                                                                                                                                                                                                         ‚îÇ ‚îÇ
‚îÇ ‚îÇ                             'type': 'import',                                                                                                                                                                                                                                    ‚îÇ ‚îÇ
‚îÇ ‚îÇ                             'module': alias.name,                                                                                                                                                                                                                                ‚îÇ ‚îÇ
‚îÇ ‚îÇ                             'alias': alias.asname,                                                                                                                                                                                                                               ‚îÇ ‚îÇ
‚îÇ ‚îÇ                             'line': node.lineno,                                                                                                                                                                                                                                 ‚îÇ ‚îÇ
‚îÇ ‚îÇ                             'method': 'ast'                                                                                                                                                                                                                                      ‚îÇ ‚îÇ
‚îÇ ‚îÇ                         })                                                                                                                                                                                                                                                       ‚îÇ ‚îÇ
‚îÇ ‚îÇ                 elif isinstance(node, ast.ImportFrom):                                                                                                                                                                                                                           ‚îÇ ‚îÇ
‚îÇ ‚îÇ                     module = node.module or ''                                                                                                                                                                                                                                   ‚îÇ ‚îÇ
‚îÇ ‚îÇ                     items = [alias.name for alias in node.names]                                                                                                                                                                                                                 ‚îÇ ‚îÇ
‚îÇ ‚îÇ                     imports.append({                                                                                                                                                                                                                                             ‚îÇ ‚îÇ
‚îÇ ‚îÇ                         'type': 'from_import',                                                                                                                                                                                                                                   ‚îÇ ‚îÇ
‚îÇ ‚îÇ                         'module': module,                                                                                                                                                                                                                                        ‚îÇ ‚îÇ
‚îÇ ‚îÇ                         'items': items,                                                                                                                                                                                                                                          ‚îÇ ‚îÇ
‚îÇ ‚îÇ                         'line': node.lineno,                                                                                                                                                                                                                                     ‚îÇ ‚îÇ
‚îÇ ‚îÇ                         'method': 'ast'                                                                                                                                                                                                                                          ‚îÇ ‚îÇ
‚îÇ ‚îÇ                     })                                                                                                                                                                                                                                                           ‚îÇ ‚îÇ
‚îÇ ‚îÇ         except SyntaxError as e:                                                                                                                                                                                                                                                 ‚îÇ ‚îÇ
‚îÇ ‚îÇ             print(f"    ‚ö†Ô∏è Syntax error in {file_path}: {e}")                                                                                                                                                                                                                    ‚îÇ ‚îÇ
‚îÇ ‚îÇ             self.failed_imports.append({                                                                                                                                                                                                                                         ‚îÇ ‚îÇ
‚îÇ ‚îÇ                 'file': file_path,                                                                                                                                                                                                                                               ‚îÇ ‚îÇ
‚îÇ ‚îÇ                 'error': str(e),                                                                                                                                                                                                                                                 ‚îÇ ‚îÇ
‚îÇ ‚îÇ                 'type': 'syntax_error'                                                                                                                                                                                                                                           ‚îÇ ‚îÇ
‚îÇ ‚îÇ             })                                                                                                                                                                                                                                                                   ‚îÇ ‚îÇ
‚îÇ ‚îÇ                                                                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
‚îÇ ‚îÇ         self.imports[file_path] = imports                                                                                                                                                                                                                                        ‚îÇ ‚îÇ

‚îÇ ‚îÇ     def _resolve_local_module(self, module_name):                                                                                                                                                                                                                                ‚îÇ ‚îÇ
‚îÇ ‚îÇ         """Try to resolve a module name to a local file"""                                                                                                                                                                                                                       ‚îÇ ‚îÇ
‚îÇ ‚îÇ         possible_files = [                                                                                                                                                                                                                                                       ‚îÇ ‚îÇ
‚îÇ ‚îÇ             f"{module_name}.py",                                                                                                                                                                                                                                                 ‚îÇ ‚îÇ
‚îÇ ‚îÇ             f"{module_name}/__init__.py"                                                                                                                                                                                                                                         ‚îÇ ‚îÇ
‚îÇ ‚îÇ         ]                                                                                                                                                                                                                                                                        ‚îÇ ‚îÇ
‚îÇ ‚îÇ                                                                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
‚îÇ ‚îÇ         for possible in possible_files:                                                                                                                                                                                                                                          ‚îÇ ‚îÇ
‚îÇ ‚îÇ             if possible in self.files:                                                                                                                                                                                                                                           ‚îÇ ‚îÇ
‚îÇ ‚îÇ                 return possible                                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
‚îÇ ‚îÇ                                                                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
‚îÇ ‚îÇ         # Handle imports like 'path.to.module'                                                                                                                                                                                                                                   ‚îÇ ‚îÇ
‚îÇ ‚îÇ         parts = module_name.split('.')                                                                                                                                                                                                                                           ‚îÇ ‚îÇ
‚îÇ ‚îÇ         if len(parts) > 1:                                                                                                                                                                                                                                                       ‚îÇ ‚îÇ
‚îÇ ‚îÇ             possible = f"{parts[0]}.py"                                                                                                                                                                                                                                          ‚îÇ ‚îÇ
‚îÇ ‚îÇ             if possible in self.files:                                                                                                                                                                                                                                           ‚îÇ ‚îÇ
‚îÇ ‚îÇ                 return possible                                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
‚îÇ ‚îÇ                                                                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
‚îÇ ‚îÇ         return None                                                                                                                                                                                                                                                              ‚îÇ ‚îÇ

‚îÇ ‚îÇ                                                                                                                                                                                                                                                                                  ‚îÇ ‚îÇ

‚îÇ ‚îÇ     def _check_imported_items(self, importing_file, target_file, items, import_info):                                                                                                                                                                                            ‚îÇ ‚îÇ
‚îÇ ‚îÇ         """Check if imported items exist in target file"""                                                                                                                                                                                                                       ‚îÇ ‚îÇ
‚îÇ ‚îÇ         target_content = self.files[target_file]['content']                                                                                                                                                                                                                      ‚îÇ ‚îÇ
‚îÇ ‚îÇ                                                                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
‚îÇ ‚îÇ         # Look for function/class definitions                                                                                                                                                                                                                                    ‚îÇ ‚îÇ
‚îÇ ‚îÇ         for item in items:                                                                                                                                                                                                                                                       ‚îÇ ‚îÇ
‚îÇ ‚îÇ             if item == '*':                                                                                                                                                                                                                                                      ‚îÇ ‚îÇ
‚îÇ ‚îÇ                 continue                                                                                                                                                                                                                                                         ‚îÇ ‚îÇ
‚îÇ ‚îÇ                                                                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
‚îÇ ‚îÇ             patterns = [                                                                                                                                                                                                                                                         ‚îÇ ‚îÇ
‚îÇ ‚îÇ                 rf'^def\s+{item}\s*\(',                                                                                                                                                                                                                                          ‚îÇ ‚îÇ
‚îÇ ‚îÇ                 rf'^class\s+{item}\s*[\(:]',                                                                                                                                                                                                                                     ‚îÇ ‚îÇ
‚îÇ ‚îÇ                 rf'^{item}\s*=',                                                                                                                                                                                                                                                 ‚îÇ ‚îÇ
‚îÇ ‚îÇ                 rf'^\s*{item}\s*=',                                                                                                                                                                                                                                              ‚îÇ ‚îÇ
‚îÇ ‚îÇ             ]                                                                                                                                                                                                                                                                    ‚îÇ ‚îÇ
‚îÇ ‚îÇ                                                                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
‚îÇ ‚îÇ             found = False                                                                                                                                                                                                                                                        ‚îÇ ‚îÇ
‚îÇ ‚îÇ             for pattern in patterns:                                                                                                                                                                                                                                             ‚îÇ ‚îÇ
‚îÇ ‚îÇ                 if re.search(pattern, target_content, re.MULTILINE):                                                                                                                                                                                                             ‚îÇ ‚îÇ
‚îÇ ‚îÇ                     found = True                                                                                                                                                                                                                                                 ‚îÇ ‚îÇ
‚îÇ ‚îÇ                     break                                                                                                                                                                                                                                                        ‚îÇ ‚îÇ
‚îÇ ‚îÇ                                                                                                                                                                                                                                                                                  ‚îÇ ‚îÇ
